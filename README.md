You'll need to setup up a .env file with the following contents to get this to run:

# Miro API
# Create a developer app at https://miro.com/app/settings/user-profile/apps
# Use the OAuth access token from your app's settings
MIRO_ACCESS_TOKEN=eyJ....
MIRO_CLIENT_ID=
MIRO_CLIENT_SECRET=

# Anthropic (Claude API)
# Get your key at https://console.anthropic.com/
ANTHROPIC_API_KEY=

# Delivery
# Email address to send completed workflows to
DELIVERY_EMAIL=
RESEND_API_KEY=
MIRO_EMAIL=
MIRO_PASSWORD=
RESEND_SENDER=
SLACK_BOT_TOKEN=
SLACK_APP_TOKEN=
SLACK_BOT_USER_ID=
